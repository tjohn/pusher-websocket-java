001    package org.java_websocket.exceptions;
002    
003    import org.java_websocket.framing.CloseFrame;
004    
005    public class InvalidHandshakeException extends InvalidDataException {
006    
007            /**
008             * Serializable
009             */
010            private static final long serialVersionUID = -1426533877490484964L;
011    
012            public InvalidHandshakeException() {
013                    super( CloseFrame.PROTOCOL_ERROR );
014            }
015    
016            public InvalidHandshakeException( String arg0 , Throwable arg1 ) {
017                    super( CloseFrame.PROTOCOL_ERROR, arg0, arg1 );
018            }
019    
020            public InvalidHandshakeException( String arg0 ) {
021                    super( CloseFrame.PROTOCOL_ERROR, arg0 );
022            }
023    
024            public InvalidHandshakeException( Throwable arg0 ) {
025                    super( CloseFrame.PROTOCOL_ERROR, arg0 );
026            }
027    
028    }
