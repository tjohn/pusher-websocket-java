001    package com.pusher.client;
002    
003    /**
004     * Host and port information for WebSocket end-points.
005     *
006     * General purpose Pusher clusters provided as constants, if
007     * you need to connect to a custom end-point, e.g. a test server,
008     * just construct your own and set it on your PusherOptions.
009     */
010    public class Cluster {
011    
012        public static final Cluster US_EAST = new Cluster("ws.pusherapp.com", 80, 443);
013        public static final Cluster EU = new Cluster("ws-eu.pusherapp.com", 80, 443);
014    
015        private final String host;
016        private final int wsPort;
017        private final int wssPort;
018    
019        public Cluster(final String host, final int wsPort, final int wssPort) {
020            this.host = host;
021            this.wsPort = wsPort;
022            this.wssPort = wssPort;
023        }
024    
025        public String getHost() {
026            return host;
027        }
028    
029        public int getWsPort() {
030            return wsPort;
031        }
032    
033        public int getWssPort() {
034            return wssPort;
035        }
036    }
