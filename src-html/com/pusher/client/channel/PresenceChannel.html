001package com.pusher.client.channel;
002
003import java.util.Set;
004
005import com.pusher.client.channel.User;
006
007/**
008 * An object that represents a Pusher presence channel. An implementation of
009 * this interface is returned when you call
010 * {@link com.pusher.client.Pusher#subscribePresence(String)} or
011 * {@link com.pusher.client.Pusher#subscribePresence(String, ChannelEventListener, String...)}.
012 * 
013 */
014public interface PresenceChannel extends PrivateChannel {
015
016        /**
017         * Gets a set of users currently subscribed to the channel.
018         * @return The users.
019         */
020        Set<User> getUsers();
021
022        /**
023         * Gets the user that represents the currently connected client.
024         * @return A user.
025         */
026        User getMe();
027}
