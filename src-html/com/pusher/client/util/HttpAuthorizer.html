001package com.pusher.client.util;
002
003import java.io.BufferedReader;
004import java.io.DataOutputStream;
005import java.io.IOException;
006import java.io.InputStream;
007import java.io.InputStreamReader;
008import java.net.HttpURLConnection;
009import java.net.MalformedURLException;
010import java.net.URL;
011import java.net.URLEncoder;
012import java.util.HashMap;
013
014import javax.net.ssl.HttpsURLConnection;
015
016import com.pusher.client.AuthorizationFailureException;
017import com.pusher.client.Authorizer;
018
019/**
020 * Used to authenticate a {@link com.pusher.client.channel.PrivateChannel private} or {@link com.pusher.client.channel.PresenceChannel presence}
021 * channel subscription.
022 * 
023 * <p>Makes an HTTP request to a defined HTTP endpoint. Expects an authentication token to be returned.</p>
024 * 
025 * <p>For more information see the <a href="http://pusher.com/docs/authenticating_users">Authenticating Users documentation</a>.
026 */
027public class HttpAuthorizer implements Authorizer {
028
029        private final URL endPoint;
030        private HashMap<String, String> mHeaders = new HashMap<String, String>();
031        private HashMap<String, String> mQueryStringParameters = new HashMap<String, String>();
032        private final String ENCODING_CHARACTER_SET = "UTF-8";
033
034        /**
035         * Creates a new authorizer.
036         * @param endPoint The endpoint to be called when authenticating.
037         */
038        public HttpAuthorizer(String endPoint) {
039                try {
040                        this.endPoint = Factory.newURL(endPoint);
041                } catch (MalformedURLException e) {
042                        throw new IllegalArgumentException(
043                                        "Could not parse authentication end point into a valid URL", e);
044                }
045        }
046        
047        /**
048         * Set additional headers to be sent as part of the request.
049         * @param headers
050         */
051        public void setHeaders(HashMap<String, String> headers) {
052                this.mHeaders = headers;
053        }
054        
055        /**
056         * Identifies if the HTTP request will be sent over HTTPS.
057         * @return 
058         */
059        public Boolean isSSL() {
060                return this.endPoint.getProtocol().equals("https");
061        }
062
063        /**
064         * This methods is for passing extra parameters authentication that needs to be added to query string.
065         * @param queryStringParameters the query parameters
066         */
067        public void setQueryStringParameters(HashMap<String, String> queryStringParameters) {
068                this.mQueryStringParameters = queryStringParameters;
069        }
070        
071        @Override
072        public String authorize(String channelName, String socketId)
073                        throws AuthorizationFailureException {
074
075                try {
076                        StringBuffer urlParameters = new StringBuffer(); 
077                        urlParameters.append("channel_name=").append(URLEncoder.encode(channelName, ENCODING_CHARACTER_SET));
078                        urlParameters.append("&socket_id=").append(URLEncoder.encode(socketId, ENCODING_CHARACTER_SET));
079                        
080
081                        // Adding extra parameters supplied to be added to query string.
082                        for(String parameterName : mQueryStringParameters.keySet()){
083                                urlParameters.append("&").append(parameterName).append("=");
084                                urlParameters.append(URLEncoder.encode(mQueryStringParameters.get(parameterName), ENCODING_CHARACTER_SET));
085                        }
086                        
087                        HttpURLConnection connection = null;
088                        if( this.isSSL() ) {
089                                connection = (HttpsURLConnection) endPoint.openConnection();
090                        }
091                        else {
092                                connection = (HttpURLConnection) endPoint.openConnection();
093                        }
094                        connection.setDoOutput(true);
095                        connection.setDoInput(true);
096                        connection.setInstanceFollowRedirects(false);
097                        connection.setRequestMethod("POST");
098                        connection.setRequestProperty("Content-Type",
099                                        "application/x-www-form-urlencoded");
100                        connection.setRequestProperty("charset", "utf-8");
101                        connection.setRequestProperty("Content-Length",
102                                        "" + Integer.toString(urlParameters.toString().getBytes().length));
103
104                        // Add in the user defined headers
105                        for (String headerName : mHeaders.keySet()) {
106                                String headerValue = mHeaders.get(headerName);
107                                connection.setRequestProperty(headerName, headerValue);
108                        }
109
110                        connection.setUseCaches(false);
111
112                        // Send request
113                        DataOutputStream wr = new DataOutputStream(connection.getOutputStream());
114                        wr.writeBytes(urlParameters.toString());
115                        wr.flush();
116                        wr.close();
117
118                        // Read response                        
119                        InputStream is = connection.getInputStream();
120                        BufferedReader rd = new BufferedReader(new InputStreamReader(is));
121                        String line;
122                        StringBuffer response = new StringBuffer();
123                        while ((line = rd.readLine()) != null) {
124                                response.append(line);
125                        }
126                        rd.close();
127                        
128                        int responseHttpStatus = connection.getResponseCode();
129                        if( responseHttpStatus != 200 ) {
130                                throw new AuthorizationFailureException( response.toString() );
131                        }
132
133                        return response.toString();
134
135                } catch (IOException e) {
136                        throw new AuthorizationFailureException(e);
137                }
138        }
139}
